\textbf{Question Exercise 16.3.21:}
   In a class of 60 students, 30 opted for NCC, 32 opted for NSS and 24 opted for both NCC and NSS. If one of these students is selected at random, find the probability that
                     \begin{enumerate}[label = (\roman*)]
                        \item The student opted for NCC or NSS.
                        \item The student has opted neither NCC nor NSS.
                        \item The student has opted NSS but not NCC.
                     \end{enumerate}
\textbf{Solution :}
  Let the events A, B be defined as ,\\ 
                  $A$ : "the student opted for NCC".\\
                  $B$ : "the student opted for NSS".\\
       Given that,
          \begin{align}
                  n\brak{A} &= 30,\\
                  n\brak{B} &= 32,\\
                  n\brak{A \cap B} &= 24,\\
                  n\brak{S} &= 60.
          \end{align} 
  where $S$ is set of total students(Sample Space).\\
    Then we can say that,
          \begin{align}
               \pr{A} &= \frac{n\brak{A}}{n\brak{S}} \\
                          &=\frac{1}{2}\\
               \pr{B} &= \frac{n\brak{B}}{n\brak{S}} \\
                          &= \frac{8}{15}\label{eq:8}\\
               \pr{A \cap B} &= \frac{n\brak{A \cap B}}{n\brak{S}}\\
                                     &= \frac{2}{5}\label{eq:10}
          \end{align}
    \begin{enumerate}[label = (\roman*)]
       \item The event "the student opted for NCC or NSS" can be described as "$A\cup B$",\\
               From principle of inclusion and exclusion,
                 \begin{align}
                   \pr{A\cup B} &=  \pr{A} + \pr{B} - \pr{A \cap B}\\
                                                  &= \frac{1}{2} + \frac{8}{15}  - \frac{2}{5}\\
                                                  &= \frac{19}{30}.\\
                    \therefore \pr{A \cup B} &= \frac{19}{30}.\label{eq:14} 
                 \end{align}
       \item The event "the student has opted neither NCC nor NSS" can be described as "$A^{\prime}\cap B^{\prime}$ "\\
                From Demorgan's Law, 
                 \begin{align}
                   A^{\prime}\cap B^{\prime} &=  \brak{A \cup B}^{\prime}
                 \end{align}
                So,
                 \begin{align}
	             \pr{ A^{\prime}\cap B^{\prime}} = \pr{\brak{A \cup B}^{\prime}} = 1 - \pr{A \cup B}                               
                 \end{align}
               Using \eqref{eq:14},
                 \begin{align}
                   \pr{ A^{\prime}\cap B^{\prime}} &= 1 - \frac{19}{30}\\
                                                                       &= \frac{11}{30}.\\
                    \therefore  \pr{ A^{\prime}\cap B^{\prime}} &= \frac{11}{30}
                 \end{align}
       \item The event "the student has opted NSS but not NCC" can be described as "B but not A" or "$B \cap A^{\prime}$" \\
                We know that,
                 \begin{align}
                   \pr{B \cap A^{\prime}} &= \pr{B} - \pr{ B \cap A}
                 \end{align}
               Using \eqref{eq:8} and \eqref{eq:10},
                 \begin{align}
	             \pr{B \cap A^{\prime}} &=  \frac{8}{15} -  \frac{2}{5}\\
                                                        & = \frac{2}{15}\\
                   \therefore  \pr{B \cap A^{\prime}}  & = \frac{2}{15}.
                 \end{align}
    \end{enumerate}
