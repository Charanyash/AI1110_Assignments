\textbf{Question 5(c):}
Use a graph sheet for this question.
(Take $1cm = 1unit$ along both x and y axis.)
       \begin{enumerate}[label=\roman*]
        \item Plot the following points:
               A$(0,5)$, B$(3,0)$, C$(1,0)$ and D$(1,-5)$
        \item Reflect the points B, C and D on the $y$ axis and name them as B',C' and D' respectively.
        \item Write down the coordinates of B, C' and D'.
        \item Join the points A, B, C, D, D', C', B', A in order and give a name to the closed figure ABCDD'C'B'.
        \end{enumerate}
\textbf{Solution:}
\begin{enumerate}[label=\arabic*,ref=\theenumi]
\numberwithin{equation}{enumi}
\numberwithin{figure}{enumi}
\numberwithin{table}{enumi}
    \item First we will plot the points A, B, C and D which is shown in the Fig $\ref{Fig1}$ .
    \item Now to get points of reflections of 
            \begin{align}
                  \vec{B}&=\myvec{3\\0}
             \end{align}
              \begin{align}
                  \vec{C}&=\myvec{1\\0}
              \end{align}
               \begin{align}
               \vec{D}&=\myvec{1\\-5}
               \end{align}
    about $y$ axis,we will use the below equation
                \begin{align}
              \vec{R}&= \vec{P} + 2\frac{c - \vec{n}^{\top}\vec{P}}{\norm{\vec{n}}^2}\vec{n} 
            \end{align}
          where R is the point of reflection of P about a plane \\
                 \begin{align}
                \vec{n}^{\top}\vec{x}&=c
            \end{align}
    \item
       For $y$ axis ,
          \begin{align}
           \vec{n}&=\myvec{1\\0}
           \end{align}
          \begin{align}
          c&=0
            \end{align}
\item Let us call the points of reflection of B,C,D on $y$axis as B',C',D' then
       \begin{align}
       \vec{B'}&= \vec{B} + 2\frac{c - \vec{n}^{\top}\vec{B}}{\norm{\vec{n}}^2}\vec{n}\\
                  &= \myvec{3\\0} + 2\frac{0-\myvec{1&0}\myvec{3\\0}}{\norm{\myvec{1\\0}}^2}\myvec{1\\0}\\
                  &= \myvec{3\\0} -\myvec{6\\0}\\
                  &= \myvec{-3\\0}
         \end{align}
         Similarly substituting C and D in the above equation gives us,\\
           \begin{align}
            \vec{C'}&=\myvec{-1\\0} \\
            \vec{D'}&=\myvec{-1\\-5}.
            \end{align}
  \item Now joining these points in the order of A, B, C, D, D', C',B',A which gives us a $7$ sided polygon.
     We will call it as 'ARROW HEAD'.
 \end{enumerate}
